# async/await

asyncжҳҜES7зҡ„дёҺејӮжӯҘж“ҚдҪңжңүе…ізҡ„е…ій”®еӯ—пјҢе…¶иҝ”еӣһдёҖдёӘPromiseеҜ№иұЎпјҢawaitж“ҚдҪңз¬Ұз”ЁдәҺзӯүеҫ…дёҖдёӘPromiseеҜ№иұЎ,е®ғеҸӘиғҪеңЁејӮжӯҘеҮҪж•°async functionеҶ…йғЁдҪҝз”ЁгҖӮasync/awaitзҡ„зӣ®зҡ„жҳҜз®ҖеҢ–дҪҝз”ЁеӨҡдёӘpromiseж—¶зҡ„еҗҢжӯҘиЎҢдёәпјҢе№¶еҜ№дёҖз»„Promisesжү§иЎҢжҹҗдәӣж“ҚдҪңгҖӮжӯЈеҰӮPromisesзұ»дјјдәҺз»“жһ„еҢ–еӣһи°ғпјҢasync/awaitжӣҙеғҸз»“еҗҲдәҶgeneratorsе’ҢpromisesгҖӮ

# async

# иҜӯжі•

async function name([param[, param[, ... param]]]) { statements }
  • name: еҮҪж•°еҗҚз§°гҖӮ
  • param: иҰҒдј йҖ’з»ҷеҮҪж•°зҡ„еҸӮж•°зҡ„еҗҚз§°гҖӮ
  • statements: еҮҪж•°дҪ“иҜӯеҸҘгҖӮ

asyncеҮҪж•°иҝ”еӣһдёҖдёӘPromiseеҜ№иұЎпјҢеҸҜд»ҘдҪҝз”Ёthenж–№жі•ж·»еҠ еӣһи°ғеҮҪж•°пјҢиҝ”еӣһзҡ„PromiseеҜ№иұЎдјҡиҝҗиЎҢжү§иЎҢresolveејӮжӯҘеҮҪж•°зҡ„иҝ”еӣһз»“жһңпјҢеҰӮжһңжҠӣеҮәејӮеёёеҲҷиҝҗиЎҢжӢ’з»қrejectгҖӮ

# е®һдҫӢ

async function asyncPromise(v){
    return v;
}

asyncPromise(1).then((v) => {
    console.log(v);
}).catch((e) => {
    console.log(e);
})

# await

asyncдёҖиҲ¬жҳҜй…ҚеҗҲawaitжҢҮд»ӨдҪҝз”Ёзҡ„пјҢиҜҘжҢҮд»ӨдјҡжҡӮеҒңејӮжӯҘеҮҪж•°зҡ„жү§иЎҢпјҢе№¶зӯүеҫ…Promiseжү§иЎҢпјҢз„¶еҗҺз»§з»ӯжү§иЎҢејӮжӯҘеҮҪж•°пјҢе№¶иҝ”еӣһз»“жһңгҖӮиӢҘPromiseеӨ„зҗҶејӮеёёrejectedпјҢawaitиЎЁиҫҫејҸдјҡжҠҠPromiseзҡ„ејӮеёёеҺҹеӣ жҠӣеҮәгҖӮеҸҰеӨ–пјҢеҰӮжһңawaitж“ҚдҪңз¬ҰеҗҺзҡ„иЎЁиҫҫејҸзҡ„еҖјдёҚжҳҜдёҖдёӘPromiseпјҢеҲҷиҝ”еӣһиҜҘеҖјжң¬иә«гҖӮ

# е®һдҫӢ

function promise(){
    return new Promise(function(resolve,reject){
        var rand = Math.random() * 2;
        setTimeout(function(){
            resolve(`solve ${rand}`);
        },1000)
    })
}

(async function asyncPromise(){
    var result = await promise();
    console.log(result);
})();

// жҚ•жҚүејӮеёё
function promise(){
    return new Promise(function(resolve,reject){
        throw new Error(`reject`);
    })
}

(async function asyncPromise(){
    var result = await promise();
    console.log(result);
})().catch((e) => {
    console.log(e);
});

// йқһpromiseеҜ№иұЎзӣҙжҺҘиҝ”еӣһиҜҘеҖјжң¬иә«
function notPromise(){
    return 1;
}

(async function asyncPromise(){
    var result = await notPromise();
    console.log(result);
})();
